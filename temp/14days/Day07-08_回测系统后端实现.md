```python
from sqlalchemy.orm import Session
from app.models.backtest_models import BacktestTask, BacktestResult
from datetime import datetime
import asyncio
from typing import Optional

class BacktestService:
    def __init__(self, db: Session):
        self.db = db

    def create_backtest_task(self, user_id: int, strategy_id: int,
                            name: str, start_date: datetime, end_date: datetime,
                            initial_capital: float = 100000) -> BacktestTask:
        """创建回测任务"""
        task = BacktestTask(
            user_id=user_id,
            strategy_id=strategy_id,
            name=name,
            start_date=start_date,
            end_date=end_date,
            initial_capital=initial_capital,
            status="pending"
        )

        self.db.add(task)
        self.db.commit()
        self.db.refresh(task)

        return task

    def get_task(self, task_id: int) -> Optional[BacktestTask]:
        """获取回测任务"""
        return self.db.query(BacktestTask).filter(BacktestTask.id == task_id).first()

    def list_user_tasks(self, user_id: int, status: Optional[str] = None):
        """获取用户的回测任务"""
        query = self.db.query(BacktestTask).filter(BacktestTask.user_id == user_id)
        if status:
            query = query.filter(BacktestTask.status == status)
        return query.order_by(BacktestTask.created_at.desc()).all()

    def update_progress(self, task_id: int, current_bar: int, total_bars: int):
        """更新回测进度"""
        task = self.get_task(task_id)
        if task:
            task.progress = (current_bar / total_bars * 100) if total_bars > 0 else 0
            task.current_bar = current_bar
            task.total_bars = total_bars
            self.db.commit()

    def start_backtest(self, task_id: int):
        """启动回测任务"""
        task = self.get_task(task_id)
        if task:
            task.status = "processing"
            task.started_at = datetime.now()
            self.db.commit()

    def complete_backtest(self, task_id: int, result_data: dict):
        """完成回测任务"""
        task = self.get_task(task_id)
        if task:
            task.status = "completed"
            task.completed_at = datetime.now()
            task.progress = 100.0

            # 创建结果
            result = BacktestResult(
                task_id=task_id,
                **result_data
            )

            self.db.add(result)
            self.db.commit()

    def fail_backtest(self, task_id: int, error_message: str):
        """回测失败"""
        task = self.get_task(task_id)
        if task:
            task.status = "failed"
            task.error_message = error_message
            task.completed_at = datetime.now()
            self.db.commit()

    def get_result(self, task_id: int) -> Optional[BacktestResult]:
        """获取回测结果"""
        return self.db.query(BacktestResult)\
            .filter(BacktestResult.task_id == task_id)\
            .first()
```

#### 任务1：性能指标计算（1.5小时）

创建 `app/services/metrics_calculator.py`：
```python
import numpy as np
import pandas as pd
from typing import List, Dict
import math

class MetricsCalculator:
    @staticmethod
    def calculate_returns(prices: List[float]) -> List[float]:
        """计算日收益率"""
        prices = np.array(prices)
        returns = np.diff(prices) / prices[:-1]
        return returns.tolist()

    @staticmethod
    def calculate_total_return(initial: float, final: float) -> float:
        """计算总收益率"""
        return (final - initial) / initial

    @staticmethod
    def calculate_annual_return(total_return: float, days: int) -> float:
        """计算年化收益率"""
        years = days / 365.0
        return (1 + total_return) ** (1 / years) - 1 if years > 0 else 0

    @staticmethod
    def calculate_max_drawdown(prices: List[float]) -> float:
        """计算最大回撤"""
        prices = np.array(prices)
        cumulative = np.cumprod(1 + np.diff(prices) / prices[:-1])
        running_max = np.maximum.accumulate(cumulative)
        drawdown = (cumulative - running_max) / running_max
        return float(np.min(drawdown))

    @staticmethod
    def calculate_sharpe_ratio(returns: List[float], risk_free_rate: float = 0.02) -> float:
        """计算夏普比率"""
        returns = np.array(returns)
        excess_returns = returns - risk_free_rate / 252
        return float(np.mean(excess_returns) / np.std(excess_returns) * np.sqrt(252))

    @staticmethod
    def calculate_sortino_ratio(returns: List[float], risk_free_rate: float = 0.02) -> float:
        """计算索提诺比率"""
        returns = np.array(returns)
        excess_returns = returns - risk_free_rate / 252

        downside_returns = np.where(excess_returns < 0, excess_returns, 0)
        downside_std = np.std(downside_returns)

        return float(np.mean(excess_returns) / downside_std * np.sqrt(252)) if downside_std > 0 else 0

    @staticmethod
    def calculate_calmar_ratio(total_return: float, max_drawdown: float, days: int) -> float:
        """计算卡玛比率"""
        annual_return = MetricsCalculator.calculate_annual_return(total_return, days)
        return float(annual_return / abs(max_drawdown)) if max_drawdown != 0 else 0

    @staticmethod
    def calculate_win_rate(trades: List[Dict]) -> float:
        """计算胜率"""
        if not trades:
            return 0
        winning_trades = sum(1 for t in trades if t.get('pnl', 0) > 0)
        return winning_trades / len(trades)

    @staticmethod
    def calculate_profit_factor(trades: List[Dict]) -> float:
        """计算利润因子"""
        if not trades:
            return 0

        gross_profit = sum(t.get('pnl', 0) for t in trades if t.get('pnl', 0) > 0)
        gross_loss = abs(sum(t.get('pnl', 0) for t in trades if t.get('pnl', 0) < 0))

        return gross_profit / gross_loss if gross_loss != 0 else 0
```

#### 任务2：结果聚合和对比分析（1.5小时）

创建 `app/services/result_analyzer.py`：
```python
from typing import List, Dict
from app.services.metrics_calculator import MetricsCalculator
from sqlalchemy.orm import Session
from app.models.backtest_models import BacktestResult

class ResultAnalyzer:
    @staticmethod
    def analyze_backtest(task_id: int, price_history: List[float],
                        trades: List[Dict], days: int) -> Dict:
        """分析回测结果"""

        initial_price = price_history[0]
        final_price = price_history[-1]

        # 计算基础指标
        total_return = MetricsCalculator.calculate_total_return(initial_price, final_price)
        annual_return = MetricsCalculator.calculate_annual_return(total_return, days)
        max_drawdown = MetricsCalculator.calculate_max_drawdown(price_history)
        returns = MetricsCalculator.calculate_returns(price_history)
        sharpe_ratio = MetricsCalculator.calculate_sharpe_ratio(returns)
        sortino_ratio = MetricsCalculator.calculate_sortino_ratio(returns)
        calmar_ratio = MetricsCalculator.calculate_calmar_ratio(total_return, max_drawdown, days)

        # 交易统计
        total_trades = len(trades)
        winning_trades = sum(1 for t in trades if t.get('pnl', 0) > 0)
        losing_trades = total_trades - winning_trades
        win_rate = MetricsCalculator.calculate_win_rate(trades)
        profit_factor = MetricsCalculator.calculate_profit_factor(trades)

        return {
            "total_return": total_return,
            "annual_return": annual_return,
            "max_drawdown": max_drawdown,
            "sharpe_ratio": sharpe_ratio,
            "sortino_ratio": sortino_ratio,
            "calmar_ratio": calmar_ratio,
            "total_trades": total_trades,
            "winning_trades": winning_trades,
            "losing_trades": losing_trades,
            "win_rate": win_rate,
            "profit_factor": profit_factor,
            "final_value": final_price
        }

    @staticmethod
    def compare_strategies(db: Session, strategy_ids: List[int]) -> List[Dict]:
        """对比多个策略的回测结果"""
        results = []

        for strategy_id in strategy_ids:
            result = db.query(BacktestResult)\
                .join(BacktestResult.task)\
                .filter(BacktestResult.task.strategy_id == strategy_id)\
                .first()

            if result:
                results.append({
                    "strategy_id": strategy_id,
                    "total_return": result.total_return,
                    "sharpe_ratio": result.sharpe_ratio,
                    "max_drawdown": result.max_drawdown,
                    "win_rate": result.win_rate
                })

        return results
```
