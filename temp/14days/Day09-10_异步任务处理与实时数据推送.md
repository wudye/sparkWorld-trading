#### 任务2：实现回测异步任务（1.5小时）

创建 `app/tasks/backtest_tasks.py`：
```python
from app.core.celery_app import celery_app
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker
import os
from datetime import datetime
import numpy as np
import time

DATABASE_URL = os.getenv("DATABASE_URL", "postgresql://qbot:qbot@localhost:5432/qbot")
engine = create_engine(DATABASE_URL)
SessionLocal = sessionmaker(bind=engine)

@celery_app.task(bind=True, name="backtest.run")
def run_backtest(self, task_id: int, strategy_code: str,
                 start_date: str, end_date: str, initial_capital: float):
    """
    异步执行回测任务
    """
    db = SessionLocal()

    try:
        from app.models.backtest_models import BacktestTask
        from app.services.backtest_service import BacktestService
        from app.services.result_analyzer import ResultAnalyzer

        service = BacktestService(db)

        # 更新任务状态
        service.start_backtest(task_id)

        # 模拟回测执行
        total_bars = 252  # 1年的交易日
        price_history = []
        trades = []

        # 生成模拟价格数据
        initial_price = initial_capital
        for i in range(total_bars):
            # 模拟价格变化（简化版）
            daily_return = np.random.normal(0.0005, 0.01)  # 日均涨幅 0.05%, 波动 1%
            price = initial_price * (1 + daily_return)
            price_history.append(price)
            initial_price = price

            # 更新进度
            self.update_state(
                state='PROGRESS',
                meta={'current': i + 1, 'total': total_bars}
            )
            service.update_progress(task_id, i + 1, total_bars)

            # 模拟交易
            if i % 20 == 0 and i > 0:
                trades.append({
                    'entry': price_history[i-10],
                    'exit': price,
                    'pnl': (price - price_history[i-10]) * 100,
                    'pnl_percent': (price - price_history[i-10]) / price_history[i-10]
                })

            # 模拟处理时间
            time.sleep(0.01)

        # 计算指标
        result_data = ResultAnalyzer.analyze_backtest(
            task_id=task_id,
            price_history=price_history,
            trades=trades,
            days=len(price_history)
        )

        # 添加曲线数据
        result_data['daily_returns'] = [
            float((price_history[i] - price_history[i-1]) / price_history[i-1])
            for i in range(1, len(price_history))
        ]
        result_data['equity_curve'] = [float(p) for p in price_history]

        # 计算回撤曲线
        max_val = price_history[0]
        drawdown_curve = []
        for p in price_history:
            if p > max_val:
                max_val = p
            dd = (p - max_val) / max_val
            drawdown_curve.append(float(dd))
        result_data['drawdown_curve'] = drawdown_curve

        # 保存结果
        service.complete_backtest(task_id, result_data)

        return {
            'status': 'completed',
            'task_id': task_id,
            'result': result_data
        }

    except Exception as e:
        service = BacktestService(db)
        service.fail_backtest(task_id, str(e))

        raise

    finally:
        db.close()

@celery_app.task(name="backtest.monitor")
def monitor_backtest(task_id: int):
    """监控回测任务状态"""
    from app.tasks.backtest_tasks import run_backtest

    # 获取任务状态
    result = celery_app.AsyncResult(f"backtest.run_{task_id}")

    if result.state == 'FAILURE':
        return {'status': 'failed', 'error': str(result.info)}
    elif result.state == 'SUCCESS':
        return {'status': 'completed', 'result': result.result}
    else:
        return {'status': result.state, 'progress': result.info.get('current', 0)}

@celery_app.task(bind=True, max_retries=3)
def retry_failed_backtest(self, task_id: int):
    """重试失败的回测任务"""
    try:
        # 重新提交任务
        run_backtest.delay(task_id)
    except Exception as exc:
        # 指数退避重试
        raise self.retry(exc=exc, countdown=60 * (2 ** self.request.retries))
```

#### 任务3：Redis缓存层（1小时）

创建 `app/core/cache.py`：
```python
import redis
import json
import os
from typing import Any, Optional
from functools import wraps

redis_client = redis.Redis(
    host=os.getenv("REDIS_HOST", "localhost"),
    port=int(os.getenv("REDIS_PORT", 6379)),
    db=0,
    decode_responses=True
)

class Cache:
    @staticmethod
    def get(key: str) -> Optional[Any]:
        """获取缓存"""
        value = redis_client.get(key)
        if value:
            try:
                return json.loads(value)
            except:
                return value
        return None

    @staticmethod
    def set(key: str, value: Any, ttl: int = 3600) -> bool:
        """设置缓存"""
        try:
            redis_client.setex(
                key,
                ttl,
                json.dumps(value) if isinstance(value, (dict, list)) else value
            )
            return True
        except:
            return False

    @staticmethod
    def delete(key: str) -> bool:
        """删除缓存"""
        return redis_client.delete(key) > 0

    @staticmethod
    def exists(key: str) -> bool:
        """检查缓存是否存在"""
        return redis_client.exists(key) > 0

def cache_key(*args, **kwargs):
    """生成缓存key"""
    parts = [str(arg) for arg in args]
    parts.extend([f"{k}:{v}" for k, v in kwargs.items()])
    return ":".join(parts)

def cached(ttl: int = 3600, key_func=None):
    """缓存装饰器"""
    def decorator(func):
        @wraps(func)
        async def async_wrapper(*args, **kwargs):
            # 生成cache key
            if key_func:
                key = key_func(*args, **kwargs)
            else:
                key = cache_key(func.__name__, *args, **kwargs)

            # 尝试从缓存获取
            cached_value = Cache.get(key)
            if cached_value is not None:
                return cached_value

            # 执行函数
            result = await func(*args, **kwargs)

            # 存入缓存
            Cache.set(key, result, ttl)

            return result

        @wraps(func)
        def sync_wrapper(*args, **kwargs):
            if key_func:
                key = key_func(*args, **kwargs)
            else:
                key = cache_key(func.__name__, *args, **kwargs)

            cached_value = Cache.get(key)
            if cached_value is not None:
                return cached_value

            result = func(*args, **kwargs)
            Cache.set(key, result, ttl)

            return result

        # 返回合适的wrapper
        import asyncio
        if asyncio.iscoroutinefunction(func):
            return async_wrapper
        else:
            return sync_wrapper

    return decorator
```

```python
from fastapi import APIRouter, Depends, Query, Path

@router.post("/tasks/{task_id}/execute-async")

        end_date=task.end_date.isoformat(),

    return {
        "task_id": task_id,
        "celery_task_id": celery_task.id,
        "status": "submitted"
    }

async def get_task_status(
    task_id: int = Path(..., ge=1)

    """获取异步任务状态"""
    # 这里应该存储celery_task_id和task_id的映射
    # 简化版本：直接查询数据库

    return {
        "task_id": task_id,
        "status": "running",
        "progress": 45
    }

@router.get("/celery-tasks/{celery_task_id}")
async def get_celery_task_status(celery_task_id: str):
    """获取Celery任务状态"""
    result = celery_app.AsyncResult(celery_task_id)

    if result.state == 'PENDING':
        response = {
            'state': 'pending',
            'current': 0,
            'total': 100,
            'status': 'Waiting...'
        }
    elif result.state == 'PROGRESS':
        response = {
            'state': 'progress',
            'current': result.info.get('current', 0),
            'total': result.info.get('total', 100),
            'status': result.info.get('status', '')
        }
    elif result.state == 'SUCCESS':
        response = {
            'state': 'success',
            'current': 100,
            'total': 100,
            'status': 'Task completed!',
            'result': result.result
        }
    elif result.state == 'FAILURE':
        response = {
            'state': 'failure',
            'current': 0,
            'total': 100,
            'status': 'Task failed',
            'error': str(result.info)
        }
    else:
        response = {
            'state': result.state,
            'current': 0,
            'total': 100,
            'status': 'Unknown'
        }

    return response
```

```python
from fastapi import APIRouter, WebSocket, WebSocketDisconnect, Query
from sqlalchemy.orm import Session
from app.core.database import get_db
from app.core.cache import Cache
import json
import asyncio
from typing import Set
from datetime import datetime

router = APIRouter()

class ConnectionManager:
    def __init__(self):
        self.active_connections: dict[str, Set[WebSocket]] = {}

    async def connect(self, websocket: WebSocket, client_id: str):
        await websocket.accept()
        if client_id not in self.active_connections:
            self.active_connections[client_id] = set()
        self.active_connections[client_id].add(websocket)

    def disconnect(self, websocket: WebSocket, client_id: str):
        if client_id in self.active_connections:
            self.active_connections[client_id].discard(websocket)
            if not self.active_connections[client_id]:
                del self.active_connections[client_id]

    async def broadcast(self, client_id: str, message: dict):
        if client_id in self.active_connections:
            for connection in list(self.active_connections[client_id]):
                try:
                    await connection.send_json(message)
                except Exception as e:
                    print(f"Error sending message: {e}")

manager = ConnectionManager()

@router.websocket("/ws/market/{symbol}")
async def websocket_market_feed(websocket: WebSocket, symbol: str):
    """市场行情实时推送"""
    client_id = f"market_{symbol}"
    await manager.connect(websocket, client_id)

    try:
        while True:
            # 接收客户端消息
            data = await websocket.receive_text()
            message = json.loads(data)

            if message.get("action") == "subscribe":
                # 订阅消息
                await websocket.send_json({
                    "type": "subscribed",
                    "symbol": symbol,
                    "timestamp": datetime.now().isoformat()
                })

            # 模拟实时行情推送
            await asyncio.sleep(2)

            # 从缓存或数据库获取最新行情
            market_data = Cache.get(f"market:{symbol}") or {
                "symbol": symbol,
                "price": 100.0 + (hash(symbol) % 50),
                "bid": 99.5,
                "ask": 100.5,
                "volume": 1000000,
                "timestamp": datetime.now().isoformat()
            }

            await websocket.send_json({
                "type": "market_data",
                "data": market_data
            })

    except WebSocketDisconnect:
        manager.disconnect(websocket, client_id)

@router.websocket("/ws/backtest/{task_id}")
async def websocket_backtest_progress(websocket: WebSocket, task_id: int):
    """回测进度实时推送"""
    client_id = f"backtest_{task_id}"
    await manager.connect(websocket, client_id)

    try:
        while True:
            # 获取回测进度
            progress_key = f"backtest:progress:{task_id}"
            progress = Cache.get(progress_key) or {}

            # 发送进度更新
            if progress:
                await websocket.send_json({
                    "type": "progress",
                    "data": progress
                })

            await asyncio.sleep(1)

    except WebSocketDisconnect:
        manager.disconnect(websocket, client_id)

@router.websocket("/ws/trades")
async def websocket_trade_updates(websocket: WebSocket, user_id: int = Query(...)):
    """交易更新实时推送"""
    client_id = f"trades_{user_id}"
    await manager.connect(websocket, client_id)

    try:
        while True:
            # 获取用户最新交易
            trades_key = f"trades:{user_id}"
            trades = Cache.get(trades_key) or []

            if trades:
                await websocket.send_json({
                    "type": "trade_update",
                    "data": trades[-1]  # 发送最新交易
                })

            await asyncio.sleep(2)

    except WebSocketDisconnect:
        manager.disconnect(websocket, client_id)
```
