```python
import time

@pytest.mark.performance
def test_list_strategies_performance(client, auth_headers, db):
    """性能测试：列表查询"""
    from app.models.models import User, Strategy

    user = db.query(User).filter(User.username == "testuser").first()

    # 创建100个策略
    for i in range(100):
        strategy = Strategy(
            user_id=user.id,
            name=f"Strategy {i}",
            code="def run(self): pass"
        )
        db.add(strategy)
    db.commit()

    # 测试性能
    start_time = time.time()

    for _ in range(10):
        response = client.get(
            f"/api/v1/strategies/user/{user.id}?limit=10",
            headers=auth_headers
        )
        assert response.status_code == 200

    elapsed = time.time() - start_time

    # 应该在2秒内完成10次查询
```

```bash
set -e

# 5. 初始化数据
echo "Initializing database..."
docker-compose -f docker-compose.prod.yml exec -T api python app/core/init_db.py

# 6. 健康检查
echo "Waiting for services to be healthy..."
for i in {1..30}; do
    if curl -f http://localhost:8000/health > /dev/null 2>&1; then
        echo "✓ API is healthy"
        break
    fi
    echo "Waiting... ($i/30)"
    sleep 1
done

echo "=== Deployment completed successfully ==="
```

```python
        pool_pre_ping=True,
        query_cache_size=500,
        echo=False,
        connect_args={
            "timeout": 30,
            "check_same_thread": False if "sqlite" in database_url else None
        }
    )

    # 监听连接事件
    @event.listens_for(engine, "connect")

        """优化连接配置"""
        if "postgresql" in database_url:
            dbapi_conn.set_isolation_level(1)  # AUTOCOMMIT
        elif "sqlite" in database_url:
            dbapi_conn.execute("PRAGMA foreign_keys=ON")

    return engine
```

```python
    ['method', 'endpoint']
)

    'active_backtest_tasks',
    'Number of active backtest tasks'

    'database_pool_size',
    'Database connection pool size'
)

# Prometheus中间件
from fastapi import Request
from starlette.middleware.base import BaseHTTPMiddleware

class PrometheusMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        start_time = time.time()

        try:
            response = await call_next(request)
        except Exception as exc:
            status_code = 500
            raise
        else:
            status_code = response.status_code

        duration = time.time() - start_time

        http_requests_total.labels(
            method=request.method,
            endpoint=request.url.path,
            status_code=status_code
        ).inc()

        http_request_duration_seconds.labels(
            method=request.method,
            endpoint=request.url.path
        ).observe(duration)

        return response
```
